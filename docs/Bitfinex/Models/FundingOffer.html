def to_new_order_packet
  data = {
    type: @type,
    symbol: @symbol,
    amount: BigDecimal(@amount, 8).to_s,
    rate: BigDecimal(@rate, 8).to_s,
    period: 2
  }

  data[:flags] = @flags unless @flags.nil?
  data
end
