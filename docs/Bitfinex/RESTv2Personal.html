def alert(price, symbol = 'tBTCUSD', type = 'price')
  params = {
    type: type,
    price: price,
    symbol: symbol
  }
  authenticated_post('auth/w/alert/set', params: params).body
end

def available_balance(rate, dir, type, symbol)
  params = {
    symbol: symbol,
    dir: dir,
    type: type,
    rate: rate
  }
  authenticated_post('auth/calc/order/avail', params: params).body
end
